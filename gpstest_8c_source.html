74 void UART_init(uint8_t uart_num, uint32_t baudrate)
75 {
76  check_uart_num(uart_num,);
77  UART *uart = &UART_uarts[uart_num];
78 
79  /* Check UART is initialised */
80  if(!uart->initialised)
81  {
82  /* Initialise UART Peripheral if not already initialised */
83  if(!SysCtlPeripheralReady(uart->peripheral_uart))
84  {
85  SysCtlPeripheralEnable(uart->peripheral_uart);
86  while(!SysCtlPeripheralReady(uart->peripheral_uart)) { };
87  }
88 
89  /* Initialise GPIO Peripheral if not already initialised */
90  if(!SysCtlPeripheralReady(uart->peripheral_gpio))
91  {
92  SysCtlPeripheralEnable(uart->peripheral_gpio);
93  while(!SysCtlPeripheralReady(uart->peripheral_gpio)) { };
94  }
95 
96  GPIOPinConfigure(uart->pin_rx_function);
97  GPIOPinConfigure(uart->pin_tx_function);
98  GPIOPinTypeUART(uart->base_gpio, uart->pin_rx | uart->pin_tx);
99 
100  UARTClockSourceSet(uart->base_uart, UART_CLOCK_SYSTEM);
101 
102  uart->initialised = true;
103  }
104 
105  UARTConfigSetExpClk(uart->base_uart, SysCtlClockGet(), baudrate,
106  (UART_CONFIG_WLEN_8 | UART_CONFIG_STOP_ONE | UART_CONFIG_PAR_NONE));
107 }
