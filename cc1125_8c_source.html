63 * @fn radio_set_freq_f
64 *
65 * @brief Sets the frequency to the requested floating point value. Will
66 * overwrite the input value with the actual value. Call only when
67 * in 'idle' state
68 *
69 * @param radio_id - select the radio to use
70 * *freq - pointer to the requested frequency
71 *
72 * @return 0 - changed sucessfully; 1 - error, freq not set
73 */
74 uint8_t radio_set_freq_f(uint8_t radio_id, double *freq){
75 
76  uint8_t div, bandsel;
77  if ((*freq >= 136.7) && (*freq <= 160)){
78  div = 24;
79  bandsel = 11; }
80  else if ((*freq >= 164) && (*freq <= 192)){
81  div = 20;
82  bandsel = 10; }
83  else if ((*freq >= 205) && (*freq <= 240)){
84  div = 16;
85  bandsel = 8; }
86  else if ((*freq >= 273.3) && (*freq <= 320)){
87  div = 12;
88  bandsel = 6; }
89  else if ((*freq >= 410) && (*freq <= 480)){
90  div = 8;
91  bandsel = 4; }
92  else if ((*freq >= 820) && (*freq <= 960)){
93  div = 4;
94  bandsel = 2; }
95  else
96  return 1;
97 
98  double f;
99  f = div * ((*freq)*1000000) * 65536 / CC_XO_FREQ;
100 
101  uint32_t freq_reg = (uint32_t)f;
102 
103  // work back to calculate the actual freq
104  f = (double)freq_reg*CC_XO_FREQ;
105  f = f / div;
106  f = f / 65536;
107 
108  *freq = f/1000000;
109 
110 
111  //now write to CC112X
112  uint8_t writebyte;
113  writebyte = freq_reg & 0xFF;
114  cc112xSpiWriteReg(radio_id, CC112X_FREQ0, &writebyte);
115  writebyte = (freq_reg >> 8) & 0xFF;
116  cc112xSpiWriteReg(radio_id, CC112X_FREQ1, &writebyte);
117  writebyte = (freq_reg >> 16) & 0xFF;
118  cc112xSpiWriteReg(radio_id, CC112X_FREQ2, &writebyte);
119  writebyte = 0x10 | bandsel;
120  cc112xSpiWriteReg(radio_id, CC112X_FS_CFG, &writebyte);
121 
122 
123  return 0;
124

151 * @fn manualCalibration
152 *
153 * @brief Calibrates radio according to CC112x errata
154 *
155 * @param none
156 *
157 * @return none
158 */
159 #define VCDAC_START_OFFSET 2
160 #define FS_VCO2_INDEX 0
161 #define FS_VCO4_INDEX 1
162 #define FS_CHP_INDEX 2
163 void manualCalibration(uint8_t radio_id) {
164 
165  uint8_t original_fs_cal2;
166  uint8_t calResults_for_vcdac_start_high[3];
167  uint8_t calResults_for_vcdac_start_mid[3];
168  uint8_t marcstate;
169  uint8_t writeByte;
170 
171  // 1) Set VCO cap-array to 0 (FS_VCO2 = 0x00)
172  writeByte = 0x00;
173  cc112xSpiWriteReg(radio_id, CC112X_FS_VCO2, &writeByte);
174 
175  // 2) Start with high VCDAC (original VCDAC_START + 2):
176  cc112xSpiReadReg(radio_id, CC112X_FS_CAL2, &original_fs_cal2);
177  writeByte = (uint8_t)(original_fs_cal2 + VCDAC_START_OFFSET);
178  cc112xSpiWriteReg(radio_id, CC112X_FS_CAL2, &writeByte);
179 
180  // 3) Calibrate and wait for calibration to be done
181  // (radio back in IDLE state)
182  SPI_cmdstrobe(radio_id, CC112X_SCAL);
183 
184  do {
185  cc112xSpiReadReg(radio_id, CC112X_MARCSTATE, &marcstate);
186  } while (marcstate != 0x41);
187 
188  // 4) Read FS_VCO2, FS_VCO4 and FS_CHP register obtained with
189  // high VCDAC_START value
190  cc112xSpiReadReg(radio_id, CC112X_FS_VCO2,
191  &calResults_for_vcdac_start_high[FS_VCO2_INDEX]);
192  cc112xSpiReadReg(radio_id, CC112X_FS_VCO4,
193  &calResults_for_vcdac_start_high[FS_VCO4_INDEX]);
194  cc112xSpiReadReg(radio_id, CC112X_FS_CHP,
195  &calResults_for_vcdac_start_high[FS_CHP_INDEX]);
196 
197  // 5) Set VCO cap-array to 0 (FS_VCO2 = 0x00)
198  writeByte = 0x00;
199  cc112xSpiWriteReg(radio_id, CC112X_FS_VCO2, &writeByte);
200 
201  // 6) Continue with mid VCDAC (original VCDAC_START):
202  writeByte = original_fs_cal2;
203  cc112xSpiWriteReg(radio_id, CC112X_FS_CAL2, &writeByte);
204 
205  // 7) Calibrate and wait for calibration to be done
206  // (radio back in IDLE state)
207 
208  SPI_cmdstrobe(radio_id, CC112X_SCAL);
209 
210  do {
211  cc112xSpiReadReg(radio_id, CC112X_MARCSTATE, &marcstate);
212  } while (marcstate != 0x41);
213 
214  // 8) Read FS_VCO2, FS_VCO4 and FS_CHP register obtained
215  // with mid VCDAC_START value
216  cc112xSpiReadReg(radio_id, CC112X_FS_VCO2,
217  &calResults_for_vcdac_start_mid[FS_VCO2_INDEX]);
218  cc112xSpiReadReg(radio_id, CC112X_FS_VCO4,
219  &calResults_for_vcdac_start_mid[FS_VCO4_INDEX]);
220  cc112xSpiReadReg(radio_id, CC112X_FS_CHP,
221  &calResults_for_vcdac_start_mid[FS_CHP_INDEX]);
222 
223  // 9) Write back highest FS_VCO2 and corresponding FS_VCO
224  // and FS_CHP result
225 
226  if (calResults_for_vcdac_start_high[FS_VCO2_INDEX] >
227  calResults_for_vcdac_start_mid[FS_VCO2_INDEX]) {
228  writeByte = calResults_for_vcdac_start_high[FS_VCO2_INDEX];
229  cc112xSpiWriteReg(radio_id, CC112X_FS_VCO2, &writeByte);
230  writeByte = calResults_for_vcdac_start_high[FS_VCO4_INDEX];
231  cc112xSpiWriteReg(radio_id, CC112X_FS_VCO4, &writeByte);
232  writeByte = calResults_for_vcdac_start_high[FS_CHP_INDEX];
233  cc112xSpiWriteReg(radio_id, CC112X_FS_CHP, &writeByte);
234  } else {
235  writeByte = calResults_for_vcdac_start_mid[FS_VCO2_INDEX];
236  cc112xSpiWriteReg(radio_id, CC112X_FS_VCO2, &writeByte);
237  writeByte = calResults_for_vcdac_start_mid[FS_VCO4_INDEX];
238  cc112xSpiWriteReg(radio_id, CC112X_FS_VCO4, &writeByte);
239  writeByte = calResults_for_vcdac_start_mid[FS_CHP_INDEX];
240  cc112xSpiWriteReg(radio_id, CC112X_FS_CHP, &writeByte);
241  }
242 }
